vectorize_sequences <- function(sequences, dimension = words_max) {
  results <- matrix(0, nrow = length(sequences), ncol = dimension)
  for (i in 1:length(sequences)){
    results[i, sequences[[i]]] <- 1
  }
  return(results)
}
